## 3. 添加新的 Composable

### 3.1 创建 Composable 文件

在 `src/composables/` 目录下创建新文件：

```typescript
// src/composables/useNewFeature.ts
import { ref, computed } from 'vue'
import { logger } from '@/utils/logger'

/**
 * 新功能 Composable
 * 提供新功能的逻辑和状态管理
 *
 * @returns 新功能的状态和方法
 */
export function useNewFeature() {
  // ========== 状态 ==========
  const isActive = ref(false)
  const data = ref<string[]>([])

  // ========== 计算属性 ==========
  const dataCount = computed(() => data.value.length)

  // ========== 方法 ==========

  /**
   * 激活功能
   */
  function activate(): void {
    isActive.value = true
    logger.info('功能已激活')
  }

  /**
   * 添加数据
   * @param item - 要添加的数据项
   */
  function addData(item: string): void {
    data.value.push(item)
  }

  // ========== 返回 ==========
  return {
    // 状态
    isActive,
    data,

    // 计算属性
    dataCount,

    // 方法
    activate,
    addData,
  }
}
```

**最佳实践**:

- 使用 `use` 前缀命名
- 添加详细的 JSDoc 注释
- 单一职责，避免过于复杂
- 返回对象包含状态和方法
- 使用 TypeScript 类型注解

### 3.2 编写单元测试

在 `src/composables/__tests__/` 目录下创建测试文件：

```typescript
// src/composables/__tests__/useNewFeature.test.ts
import { describe, it, expect } from 'vitest'
import { useNewFeature } from '../useNewFeature'

describe('useNewFeature', () => {
  it('应该正确初始化状态', () => {
    const { isActive, data } = useNewFeature()

    expect(isActive.value).toBe(false)
    expect(data.value).toEqual([])
  })

  it('应该正确激活功能', () => {
    const { isActive, activate } = useNewFeature()

    activate()

    expect(isActive.value).toBe(true)
  })

  it('应该正确添加数据', () => {
    const { data, addData, dataCount } = useNewFeature()

    addData('item1')
    addData('item2')

    expect(data.value).toEqual(['item1', 'item2'])
    expect(dataCount.value).toBe(2)
  })
})
```

### 3.3 在组件中使用

  activate()

## 5. 更新游戏数据

### 5.1 使用 PathResolver 直接更新

```typescript
import { PathResolver } from '@/services/dataUpdate'
import { useGameStore } from '@/stores/gameStore'

const gameStore = useGameStore()

// 更新单个字段
PathResolver.setValueByPath(gameStore.gameData, 'shop.currency', 1000)

// 删除字段
PathResolver.deleteByPath(gameStore.gameData, 'system.storage.inventory.used_item')
```

### 5.2 使用 DataMerger 应用 YAML 更新

```typescript
import { DataMerger } from '@/services/dataUpdate'
import { useGameStore } from '@/stores/gameStore'

const gameStore = useGameStore()

// 应用 YAML 更新（支持 $update 和 $delete 操作符）
const yamlContent = `
$update:
  "gameData.shop.currency": 1000
  "gameData.characters.player.level": 2

$delete:
  - "gameData.system.storage.inventory.used_item"
`

DataMerger.applyYamlUpdate(gameStore.gameData, yamlContent)
```

### 5.3 直接修改响应式数据（推荐用于简单更新）

```typescript
import { useGameStore } from '@/stores/gameStore'

const gameStore = useGameStore()

// 直接修改（Vue 3 响应式系统会自动追踪）
gameStore.gameData.shop.currency = 1000
gameStore.gameData.characters.player.level = 2
```

### 5.4 数据路径规范

数据路径使用点分隔的字符串：

```typescript
// 正确的路径示例
'shop.currency'
'characters.player.level'
'characters.player.psychologicalState.content.理智.current'
'achievements.ach_001.unlocked'

// 在 YAML 更新中需要加 "gameData." 前缀
'gameData.shop.currency'
'gameData.characters.player.level'

// 错误的路径示例
'shop/currency' // 不要使用斜杠
'characters[player].level' // 不要使用方括号
```

## 9. 性能优化

### 9.1 避免不必要的计算

使用 `computed` 缓存计算结果：

```typescript
// ❌ 不好：每次访问都重新计算
const totalPrice = () => {
  return items.value.reduce((sum, item) => sum + item.price, 0)
}

// ✅ 好：使用 computed 缓存
const totalPrice = computed(() => {
  return items.value.reduce((sum, item) => sum + item.price, 0)
})
```

### 9.2 使用防抖

对于频繁触发的事件使用防抖：

```typescript
import { debounce } from '@/utils/debounce'

const handleSearch = debounce((query: string) => {
  // 搜索逻辑
}, 300)
```

### 9.3 批量更新

批量更新数据而非多次单独更新：

```typescript
// ❌ 不好：多次更新
gameStore.updateGameData('shop.currency', 1000)
gameStore.updateGameData('characters.player.level', 2)
gameStore.updateGameData('characters.player.exp', 0)

// ✅ 好：批量更新
gameStore.batchUpdateGameData([
  { path: 'shop.currency', value: 1000 },
  { path: 'characters.player.level', value: 2 },
  { path: 'characters.player.exp', value: 0 },
])
```

### 10.2 显示错误提示

```typescript
import { useToast } from '@/composables/useToast'

const { error: showErrorToast } = useToast()

try {
  // 操作
} catch (error) {
  if (error instanceof AppError) {
    showErrorToast(error.getUserMessage())
  } else {
    showErrorToast('操作失败，请稍后重试')
  }
}
```

## 12. 常见问题

**Q: 如何添加新的数据字段？**

A:

1. 在 `src/types/gameData.ts` 中扩展 `GameData` 接口
2. 在 YAML 数据中添加字段
3. 更新相关的组件和 Composables

**Q: 如何处理异步操作？**

A: 使用 `async/await` 并添加错误处理：

```typescript
async function loadData() {
  try {
    const data = await fetchData()
    // 处理数据
  } catch (error) {
    logger.error('加载失败:', error)
    showErrorToast('加载失败')
  }
}
```
